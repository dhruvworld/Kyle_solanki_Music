# Build and train FFNN for behavioral skip prediction

input_dim = X_train_np.shape[1]
print('Input dimension:', input_dim)

model = tfk.Sequential([
    tfl.Input(shape=(input_dim,)),
    tfl.Dense(32, activation='relu'),
    tfl.Dropout(0.3),
    tfl.Dense(16, activation='relu'),
    tfl.Dense(1, activation='sigmoid'),
])

model.compile(
    optimizer=tfk.optimizers.Adam(learning_rate=1e-3),
    loss=tfk.losses.BinaryCrossentropy(),
    metrics=[tfk.metrics.BinaryAccuracy(name='accuracy')]
)

model.summary()

# Early stopping on validation loss (using validation_split)
callbacks = [
    tfkc.EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)
]

history = model.fit(
    X_train_np, y_train,
    validation_split=0.1,
    epochs=100,
    batch_size=256,
    verbose=1,
    callbacks=callbacks
)

class BehaviorDataset(Dataset):
    def __init__(self, X, y):
        self.X = torch.tensor(X, dtype=torch.float32)
        self.y = torch.tensor(y, dtype=torch.float32).view(-1, 1)

    def __len__(self):
        return self.X.shape[0]

    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]

batch_size = 256

# Define FFNN in PyTorch

input_dim = X_all.shape[1]
print('PyTorch model input_dim =', input_dim)

class FFNN(nn.Module):
    def __init__(self, input_dim):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, 32),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(32, 16),
            nn.ReLU(),
            nn.Linear(16, 1),
            nn.Sigmoid(),
        )

    def forward(self, x):
        return self.net(x)

model_pt = FFNN(input_dim)
criterion = nn.BCELoss()
optimizer = torch.optim.Adam(model_pt.parameters(), lr=1e-3)

# Training loop

num_epochs = 30
train_losses = []
val_losses = []

for epoch in range(1, num_epochs + 1):
    model_pt.train()
    running_loss = 0.0
    for xb, yb in train_loader:
        optimizer.zero_grad()
        preds = model_pt(xb)
        loss = criterion(preds, yb)
        loss.backward()
        optimizer.step()
        running_loss += loss.item() * xb.size(0)
    epoch_train_loss = running_loss / len(train_dataset)

    # Validation
    model_pt.eval()
    val_running_loss = 0.0
    with torch.no_grad():
        for xb, yb in val_loader:
            preds = model_pt(xb)
            loss = criterion(preds, yb)
            val_running_loss += loss.item() * xb.size(0)
    epoch_val_loss = val_running_loss / len(val_dataset)

    train_losses.append(epoch_train_loss)
    val_losses.append(epoch_val_loss)

    print(f"Epoch {epoch:03d}: train_loss={epoch_train_loss:.4f}, val_loss={epoch_val_loss:.4f}")

FFNN Architecture Summary

We use the same 4-layer feedforward architecture in both Keras and PyTorch:

  • Input layer: all scaled behavioral + Spotify features (spotify_popularity, is_explicit, album_release_year, tempo_bpm_synth, one-hot time_of_day_synth_*).
  • Hidden layer 1: Dense/Linear with 32 units, ReLU activation, followed by Dropout(0.3).
  • Hidden layer 2: Dense/Linear with 16 units, ReLU activation.
  • Output layer: 1 unit with sigmoid activation for binary classification (skipped_synth {0,1}).

This matches the project requirement of a simple FFNN for binary classification and is easy to extend in later experiments (more layers, different widths, different dropout, etc.).

Early Model Behavior and Planned Improvements

Early behavior observations

  • Both FFNNs (Keras and PyTorch) train on a heavily imbalanced label distribution where only about ~910% of examples have skipped_synth = 1.
  • As a result, early models tend to predict "not skipped" for most examples:
    • Accuracy looks high, but recall for the skipped class is very low and ROC AUC is near 0.5.
    • This shows that accuracy alone is misleading for this task; we must inspect confusion matrices and recall/precision for the positive class.

Planned improvements

  • Handle class imbalance: Try class weighting, focal loss, or oversampling/undersampling to boost recall for skipped_synth = 1.
  • Feature engineering: Incorporate text-based features from tags (e.g., TFIDF or embeddings) and possibly additional Spotify metadata such as genre or popularity buckets.
  • Hyperparameter tuning: Systematically vary hidden layer sizes, dropout rates, and learning rates (or use grid/random search) and track validation ROC AUC.
  • Model comparison: Add simpler baselines (logistic regression, calibrated Random Forest) and compare against the FFNNs.
  • Evaluation focus: Report and optimize for metrics that matter for skipping behavior (ROC AUC, precision/recall for class 1, maybe PR curves), not just overall accuracy.

# Threshold tuning for improved model (optimize F1 for skipped class)
from sklearn.metrics import precision_recall_fscore_support

probs_imp = improved_model.predict(X_test_np).ravel()

best_thresh = 0.5
best_f1 = -1.0
best_stats = None
thresholds = np.linspace(0.1, 0.9, 17)

for thr in thresholds:
    preds_thr = (probs_imp >= thr).astype(int)
    precision, recall, f1, _ = precision_recall_fscore_support(
        y_test, preds_thr, average='binary', zero_division=0
    )
    if f1 > best_f1:
        best_f1 = f1
        best_thresh = thr
        best_stats = (precision, recall, f1)

print(f'Best threshold (by F1 for skipped class): {best_thresh:.2f}')
print(f'Precision: {best_stats[0]:.3f}, Recall: {best_stats[1]:.3f}, F1: {best_stats[2]:.3f}')

# Confusion matrix at best threshold
best_preds = (probs_imp >= best_thresh).astype(int)
cm_imp = confusion_matrix(y_test, best_preds)
ConfusionMatrixDisplay(cm_imp, display_labels=['Not skipped', 'Skipped']).plot()
plt.title(f'Improved Keras FFNN Confusion Matrix (thr={best_thresh:.2f})')
plt.show()
